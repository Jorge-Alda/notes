# $\mu_a \neq 0$
## $c_G \neq 0$, $c_q^A = 0$
$$m_a^2 = \mu_a^2 + \frac{2}{3}c_G^2\frac{F_\pi^2}{f_a^2}m_0^2 \left(1 + \frac{m_0^2}{\mu_a^2}\right)$$
## $c_G = 0$, $c_{uu}^A \neq 0$
$$\begin{align}
m_a^2 = \mu_a^2 + \frac{{c_{uu}^A}^2}{12} \frac{F_\pi^2}{f_a^2}
&\Bigg[ 3 m_{\pi^0}^2 + m_\eta^2(c_\eta-\sqrt{2}s_\eta)^2+m_{\eta'}^2(\sqrt{2}c_\eta + s_\eta)^2\\ 
&-2\sqrt{3}m_{\pi^0}^2[(c_\eta-\sqrt{2}s_\eta)\theta_{\pi^0\eta} + (\sqrt{2}c_\eta +s_\eta)\theta_{\pi^0\eta'}] + 2\sqrt{3}m_\eta^2(c_\eta-\sqrt{2}s_\eta)\theta_{\pi^0\eta} + 2\sqrt{3}m_{\eta'}^2(\sqrt{2}c_\eta+s_\eta)\theta_{\pi^0\eta'}\\
&+ \frac{3 m_{\pi^0}^4 + m_\eta^4(c_\eta-\sqrt{2}s_\eta)^2+m_{\eta'}^4(\sqrt{2}c_\eta + s_\eta)^2}{\mu_a^2}\\
&-2\sqrt{3}\frac{m_{\pi^0}^4}{\mu_a^2}[(c_\eta-\sqrt{2}s_\eta)\theta_{\pi^0\eta} + (\sqrt{2}c_\eta +s_\eta)\theta_{\pi^0\eta'}] + 2\sqrt{3}\frac{m_\eta^4}{\mu_a^2}(c_\eta-\sqrt{2}s_\eta)\theta_{\pi^0\eta} + 2\sqrt{3}\frac{m_{\eta'}^4}{\mu_a^2}(\sqrt{2}c_\eta+s_\eta)\theta_{\pi^0\eta'} \Bigg]
\end{align}$$

## $c_G=0$, $c_{dd}^A \neq 0$

$$\begin{align}
m_a^2 = \mu_a^2 + \frac{{c_{dd}^A}^2}{12}\frac{F_\pi^2}{f_a^2}&\Bigg[3 m_{\pi^0}^2  + m_\eta^2(c_\eta-\sqrt{2}s_\eta)^2+m_{\eta'}^2(\sqrt{2}c_\eta + s_\eta)^2\\ 
&+2\sqrt{3}m_{\pi^0}^2[(c_\eta-\sqrt{2}s_\eta)\theta_{\pi^0\eta} + (\sqrt{2}c_\eta +s_\eta)\theta_{\pi^0\eta'}] - 2\sqrt{3}m_\eta^2(c_\eta-\sqrt{2}s_\eta)\theta_{\pi^0\eta} - 2\sqrt{3}m_{\eta'}^2(\sqrt{2}c_\eta+s_\eta)\theta_{\pi^0\eta'}\\
&+ \frac{3 m_{\pi^0}^4 + m_\eta^4(c_\eta-\sqrt{2}s_\eta)^2+m_{\eta'}^4(\sqrt{2}c_\eta + s_\eta)^2}{\mu_a^2}\\
&+2\sqrt{3}\frac{m_{\pi^0}^4}{\mu_a^2}[(c_\eta-\sqrt{2}s_\eta)\theta_{\pi^0\eta} + (\sqrt{2}c_\eta +s_\eta)\theta_{\pi^0\eta'}] - 2\sqrt{3}\frac{m_\eta^4}{\mu_a^2}(c_\eta-\sqrt{2}s_\eta)\theta_{\pi^0\eta} - 2\sqrt{3}\frac{m_{\eta'}^4}{\mu_a^2}(\sqrt{2}c_\eta+s_\eta)\theta_{\pi^0\eta'} \Bigg]
\end{align}$$
